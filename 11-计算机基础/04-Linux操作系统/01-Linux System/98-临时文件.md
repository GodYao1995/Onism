~~~bash
# 正则查找 特殊符号需要转义 -iregex： 忽略大小写的正则 
$ find . -regex  '.*\(\.txt|\.pdf\)$'

# 否定查找
$ find . ! -name "*.pdf"

# -exec 命令用来调用并执行指定的其他命令 {} 用来接受其他命令传入的参数集合
$ find ./ -name "*.png" -exec rm {}
# 删除效率更高 -type f 文件
$ find ./ -type f -name "temp*" -delete 

# 类型定制搜索 
# -type d f l  目录、文件、链接 -maxdepth 深度
$ find . -maxdepth 1 -type f
~~~
